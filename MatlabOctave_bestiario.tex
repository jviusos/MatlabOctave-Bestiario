        \begin{fancytable}{Operaciones de ficheros}
            \cmdo{fid=fopen(\cadena{datos.txt},\cadena{r})} & Asocia la var. \cmdo{fid} al fichero \cmdo{\cadena{datos.txt}} %de la carpeta de trabajo, 
            y permiso \cmdo{\cadena{r}}.\newline Devuelve \cmdo{-1} si hay un error \\
            \cmdo{close(fid);} & Cierra el fichero asignado al nombre interno \cmdo{fid} y lo desvincula de Matlab/Octave.\newline Devuelve \cmdo{-1} si hay un error \\
            \hseparador%
            \cmdo{frewind(fid)} & ``Rebobina'' el fichero: cursor de lectoescritura al inicio\\%, es decir, coloca al inicio del fichero el cursor de lectoescritura\\
%             \hseparador%
%             \cmdo{-1} & Un \textbf{error} en apertura (o cierre) de fichero se comunica con \cmdo{fid} (o \cmdo{ans}) tomando el valor entero \cmdo{-1}\\%, \cmdo{fid} (o \cmdo{ans}) toma el valor entero \cmdo{-1}\\
%             \hseparador%
            \cmdo{feof(fid)} & Vale 1 si estamos en el final del fichero asociado a \cmdo{fid}\\
            \hseparador%
            \cmdo{fprintf(fid,\cadena{a es \%.2f}, a)} & Escribe en el fichero de nombre interno \cmdo{fid} el texto combinado con el dato de $a$ \\
%             \cmdo{fdisp(fid, A)} & Escribe en el fichero de nombre interno \cmdo{fid} el contenido de la variable $A$ \\
            \hseparador%
            \cmdo{c=fgetl(fid)} & Lee l\'inea (sin incluir '\cmdo{\verb|\n|}') del fichero \cmdo{fid} y la almacena en \cmdo{c} como {cadena}.\newline \cmdo{-1} si llega a final de fichero \\
            \cmdo{c=fgets(fid)} & Lee l\'inea (incluyendo '\cmdo{\verb|\n|}') del fichero \cmdo{fid} y la almacena en \cmdo{c} como {cadena}.\newline \cmdo{-1} si llega a final de fichero  \\
            \cmdo{c=fgets(fid,5)} & Similar, pero lee 5 caracteres siguientes a la \'ultima posici\'on, salvo final de linea.\newline \cmdo{-1} si llega a final de fichero\\
            \cmdo{[A,cont]=fscanf(fid,\cadena{\%f})} & Lee todos los datos con formato \cmdo{\cadena{\%f}} en \cmdo{fid} hasta que encuentra algo que no lo es o EOF. Los almacena en \cmdo{A} (en \underline{orden columna}), y en \cmdo{cont} el nº de datos le\'idos\\
            \cmdo{A=fscanf(fid,\cadena{\%f},2)} & Igual, pero \cmdo{A} columna $1\times 2$. Para de leer cuando rellena \cmdo{A} \\
            \cmdo{A=fscanf(fid,\cadena{\%f},[3,2])} & Igual, pero en \cmdo{A} siendo $3\times 2$, a\~nade {\bf 0}'s hasta completar\\
            \cmdo{A=fscanf(fid,\cadena{\%f},[2,inf])} & Igual, pero en \cmdo{A} con 2 filas\\
%             \cmdo{[A,cont]=sscanf(c,\cadena{\%f},2)} & Similar a \cmdo{fscanf}, pero leyendo en una cadena \cmdo{c}\\
            \hlinefinal%
        \end{fancytable}
